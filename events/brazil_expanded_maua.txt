# Barão de Mauá 
maua.1 = {	#Railway events
	type = country_event
	event_image = {
		video = "gfx/event_pictures/unspecific_trains.bk2"
		}
	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/trains"

	duration = 3
	placement = scope:maua_end
	title = maua.1.t
	desc = maua.1.d
	flavor = maua.1.f

	trigger = {

		}

	immediate = {
		random_scope_state = {
			limit = {
				state_region = s:STATE_RIO_DE_JANEIRO
			}
			save_scope_as = maua_end
		}
		ig:ig_industrialists = {
			save_scope_as = industrialists_ig
			leader = {
				save_scope_as = industrialists_leader
			}
		}
		set_global_variable = {
			name = maua.1_done
			value = yes
			}
	}
	
	option = { # estrada de ferro Maua (big boom to Rio)
		name = maua.1.a
		default_option = yes
		scope:maua_end = {
			add_modifier = {
				name = brazil_rail_mod
				months = 25
				multiplier = 2
				}
			}

		}
	option = { # Minas Triangle
		name = maua.1.b
		every_scope_state = {
			limit = {
				OR = {
					state_region = s:STATE_PARANA
					state_region = s:STATE_MINAS_GERAIS
					state_region = s:STATE_SAO_PAULO
					}
				}
			add_modifier = {
				name = brazil_rail_mod
				months = 25
				
				}
			}

		}
	}
