# The Acre Purchase succes
acre.2 = {
	type = country_event
	placement = s:STATE_ACRE.region_state:BOL

	title = acre.2.t
	desc = acre.2.d
	flavor = acre.2.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	trigger = {
	}

	immediate = {
		c:BOL = {
			save_scope_as = bolivian_scope
		}
		s:STATE_ACRE.region_state:BOL = {
			save_scope_as = amazonas_scope
		}
	}

	option = {
		name = acre.2.a
		default_option = yes
		add_modifier = {
			name = acre_brz_big_money 
			months = 7
				}
		s:STATE_AMAZONAS.region_state:BRZ = {
			add_modifier = {
				name = rubber_production_modifier
				months = 25
			}
			}

		s:STATE_ACRE.region_state:BOL = {
			set_state_owner = c:BRZ
		}
		s:STATE_AMAZONAS.region_state:BOL = {
			set_state_owner = c:BRZ
		}

		change_relations = {
			country = c:BOL
			value = 10
		}
	}
	option = { # perhaps another time
	name = acre.5.c
	c:BRZ = {
		add_modifier = {
			name = br_failed_diplo
			months = 25
			}
		}

	}
}

	type = country_event
	placement = s:STATE_ACRE.region_state:BOL

	title = acre.4.t
	desc = acre.4.d
	flavor = acre.4.f

	duration = 3

	event_image = {
		video = "gfx/event_pictures/unspecific_signed_contract.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/unspecific/signed_contract"

	icon = "gfx/interface/icons/event_icons/event_map.dds"

	trigger = {
	}

	immediate = {
		c:BOL = {
			save_scope_as = bolivian_scope
		}
		s:STATE_ACRE.region_state:BOL = {
			save_scope_as = amazonas_scope
		}
	}

	option = {
		name = acre.4.a
		ai_chance = {
			base = 1
		}
		
		add_journal_entry = {
			type = je_diplo_negotiation.2
			target = scope:bolivian_scope
		}

	}
	option = { # perhaps another time
		name = acre.4.c
		default_option = yes
		ai_chance = {
			base = 0
		}
		remove_variable = acre_purchase_var
	}
}
