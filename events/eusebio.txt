#SLAVE TRADE BANNED
namespace = eusebio_law

eusebio_law.1 = {

	type = country_event
	placement = ROOT

	title = eusebio_law.1.t
	desc = eusebio_law.1.d
	flavor = eusebio_law.1.f

	event_image = {
		video = "gfx/event_pictures/unspecific_politicians_arguing.bk2"
	}


	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/factory_opening"



	icon = "gfx/interface/icons/event_icons/event_portrait.dds"

	duration = 3

	trigger = {
		exists = c:BRZ
		c:BRZ = ROOT
		is_player = yes
		NOT = {has_global_variable = name = eusebio_enacted}
	}

	immediate = {
		random_scope_state = {
			save_scope_as = slave_insurrection_state
			}
		set_global_variable = {
			name = eusebio_enacted
			value = yes
			}

		}
	
	option = {
		name = eusebio_law.1.a
		default_option = yes
		# end slave trade
		remove_modifier = br_last_slave_country
		activate_law = law_type:law_legacy_slavery
		change_relations = {
			country = c:GBR
			value = 30
			}
		random_interest_group = {
			limit = {
				OR = {
					is_interest_group_type = ig_rural_folk
					is_interest_group_type = ig_trade_unions
					is_interest_group_type = ig_petty_bourgeoisie
					is_interest_group_type = ig_industrialists
					}
				leader = {
					NOT = { has_ideology = ideology_anti_slavery }
					}
				}
			leader = { set_ideology = ideology_anti_slavery }
		}
		remove_modifier = br_slavery_mod
		remove_modifier = br_english_raiding 
		add_modifier = {
			name = brazil_diplomatic_reward
			months = 25
		}	
	}

	option = {
		name = eusebio_law.1.b
		
		ig:ig_landowners = {
		add_modifier = {
			name = br_landowners_happy
			months = normal_modifier_time
			}
		}
		change_relations = {
			country = c:GBR
			value = -30
		}
		scope:slave_insurrection_state = {
			add_radicals = {
				value = 0.25
				pop_type = slaves
			}
		}
		add_loyalists = {
			is_pop_type = aristocrats
			value = 0.1
		}
		c:BRZ = {
			remove_modifier = br_slavery_mod  
			}

		c:BRZ = {
			remove_modifier = br_english_raiding  
			}

	}

}
