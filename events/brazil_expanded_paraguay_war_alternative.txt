				add_war_goal = { # paraguay claim Corrientes

					type = conquer_state

					}

					holder = c:ARG
					type = war_reparations
					target_country = c:PRG
						}

					}	
				}

		}

		name = par_war_alternative.1.a #OH OH FROM BR
