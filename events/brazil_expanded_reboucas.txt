namespace = reboucas_brothers

# The Rebouças brothers 
reboucas_brothers.1 = {	#Railway events
	type = country_event
	event_image = {
		video = "gfx/event_pictures/southamerica_factory_opening.bk2"
	}

	on_created_soundeffect = "event:/SFX/UI/Alerts/event_appear"
	on_opened_soundeffect = "event:/SFX/Events/southamerica/factory_opening"

	icon = "gfx/interface/icons/event_icons/event_newspaper.dds"

	duration = 3
	placement = ROOT
	title = reboucas_brothers.1.t
	desc = reboucas_brothers.1.d
	flavor = reboucas_brothers.1.f

	trigger = {
		c:BRZ = ROOT
		NOT = {has_variable = reboucas_brothers_done}
		any_scope_building = {
			is_building_type = building_railway
			level >= 3
		}
		}

	immediate = {
		set_variable ={
			name = reboucas_brothers_done
			value = yes
		}
		random_scope_state = {
			limit = {
				state_region = s:STATE_PARANA
			}
			save_scope_as = reboucas_parana
		}
	}
	
	option = { # estrada de ferro Paraná, migration to Paraná
		name = reboucas_brothers.1.a
		default_option = yes
		scope:reboucas_parana = {
			add_modifier = {
				name = brazil_rail_mod

				months = 13
				}
			add_modifier = {

				name = coop_migration_modifier
				months = 13
				}
			}

		}
		option = { #tech advance only if can research
		name = reboucas_brothers.1.b
		trigger = {
			NOT = { has_technology_researched = reinforced_concrete }
			can_research = reinforced_concrete
		}
		add_technology_progress = { progress = 4000 technology = reinforced_concrete }
	}
}
