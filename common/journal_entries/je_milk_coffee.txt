je_coffee_export = {
	icon = "gfx/interface/icons/event_icons/event_trade.dds"

	immediate = {
		random_scope_state = {
			limit = {
				OR {
				#has_building_type = building_coffee_plantation
				state_region = s:STATE_MINAS_GERAIS
				state_region = s:STATE_SAO_PAULO
				}		
			}
			save_scope_as = coffee_state
				}
		
		}
	

	possible = {

	}

	complete = {

		scope:coffee_state = {
				
				#has_modifier = coffe_and_milk_mod
				any_scope_building = {
					is_building_type = building_coffee_plantation
					level >= global_var:coffee_check
					occupancy > 0.75
					weekly_profit > 0
						}
					
					}
				
		capital.market = {
			any_trade_route = {
				exporter.owner = root
				goods = g:coffee
			}
		}
	}

	on_complete = {
		scope:coffee_state = {
			add_modifier = {
				name = coffe_and_milk_mod
				months = 24
			}
		}
		scope:target = {
			add_modifier = {
				name = ig_group_in #bonus when completing + BONUS PRODUCTION TO BE ADDED EDIT HERE
				months = 25
			}
		}


		trigger_event = {
			id = coffee_success.1
			days = 7
			popup = yes
		}
	}


	on_timeout = {
		scope:target = {
			add_modifier = {
				name = ig_group_angry #malus if not completing
				months = 13
				#ADD MORE MALUS MOD FOR TIMEOUT
			}
		}
	}

	timeout = 600

	weight = 200



	should_be_pinned_by_default = yes
}
